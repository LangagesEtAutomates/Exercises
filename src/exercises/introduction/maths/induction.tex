\begin{exercice}[Récurrence]\label{exo:introduction/maths/induction}

  \begin{question}
  \item Montrer par récurrence que pour tout entier naturel $n$ :
    $$\begin{array}{r@{~}c@{~}l}
    \displaystyle\sum_{i=1}^n (2i-1) &=& \displaystyle 1 + 3 + 5 + \dots + (2n-1)\\
    &=& \displaystyle n^2.
    \end{array}$$
  \item Montrer par récurrence que pour tout entier naturel $n$ :
    $$\begin{array}{r@{~}c@{~}l}
    \displaystyle \sum_{i=1}^n i^2 &=& \displaystyle 1 + 4 + 9 + \dots + n^2\\
    &=& \displaystyle \frac{n(n+1)(2n+1)}{6}.
    \end{array}$$
  \end{question}
  
\end{exercice}
