\begin{exercice}[Grammaires engendrant des langages algébriques]\label{exo:parsing/grammars/write_grammars}

  Donnez une grammaire algébrique engendrant chacun des langages suivants :
  \begin{question}
  \item $\{ a^n b^{2n} \mid  n > 0\}$
  \item $\{x c x^\textsc{r} \mid  x \in \{a, b\}^\star  \}$ ($x^\textsc{r}$ est le mot miroir de $x$)
  \item $\{ a^n b^m \mid  n, m \in \mathbb{N}, n \neq m \}$
  \item $\{u \in \{a, b\}^\star  \mid  |u|_a = |u|_b \}$
  \end{question}

\end{exercice}
