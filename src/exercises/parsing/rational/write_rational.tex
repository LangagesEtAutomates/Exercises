\begin{exercice}[Grammaires rationnelles]\label{exo:parsing/rational/write_rational}

  Donner une grammaire rationnelle qui engendre chacun des langages suivants.
  \begin{question}
  \item $L_1 \eqdef \mathcal{L}((a | b)^\star a b (a | b)^\star)$
  \item $L_2 \eqdef \mathcal{L}(a^\star ba | b^\star a b a)$
  \item $L_3 \eqdef \mathcal{L}\left(
    \begin{tikzpicture}[automaton, y=10mm, x=20mm, baseline=(1.base)]
      \state[initial]   (1) at (0,1) {$1$}; 
      \state            (2) at (1,2) {$2$}; 
      \state            (3) at (1,0) {$3$}; 
      \state            (4) at (2,2) {$4$}; 
      \state            (5) at (2,0) {$5$}; 
      \state[accepting] (6) at (3,1) {$6$}; 

      \path (2) edge[loop above] node       {$a$}    (2);
      \path (3) edge[loop below] node       {$b$}    (3);
      \path (4) edge[loop above] node       {$b$}    (4);
      \path (5) edge[loop below] node       {$a$}    (5);
      \path (6) edge[loop right] node       {$a, b$} (6);
      \path (1) edge             node       {$a$}    (2);
      \path (1) edge             node[swap] {$b$}    (3);
      \path (2) edge             node       {$b$}    (4);
      \path (3) edge             node[swap] {$a$}    (5);
      \path (4) edge             node       {$a$}    (6);
      \path (5) edge             node[swap] {$b$}    (6);
    \end{tikzpicture}
    \right)$
  \item $L_4 \eqdef \mathcal{L}\left(
    \begin{tikzpicture}[automaton, size=20mm, baseline=(base)]
      \coordinate (base) at (.5,.5);
      \state[initial, accepting] (1) at (0,1) {$1$}; 
      \state                     (2) at (1,1) {$2$}; 
      \state                     (3) at (1,0) {$3$}; 
      \state[accepting]          (4) at (0,0) {$4$}; 

      \path (2) edge[loop right] node {$b$}   (2);
      \path (3) edge[loop right] node {$b$}   (3);
      \path (4) edge[loop left ] node {$b$}   (4);
      \path (2) edge             node {$a$}   (3);
      \path (3) edge             node {$a$}   (4);
      \path (4) edge             node {$a,b$} (1);
      \path (3) edge             node {$b$}   (1);
      \path (1) edge[bend left]  node {$a$}   (2);
      \path (2) edge             node {$b$}   (1);
    \end{tikzpicture}
    \right)$
  \end{question}

\end{exercice}
