\begin{exercice}[Révisions -- Expressions rationnelles]\label{exo:lexing/regexp/revision}

  \begin{question}

  \item Décrire les langages définis par les expressions suivantes (sur les alphabets indiqués) :
    \begin{enumerate}
    \item $\mathit{regarder}(ai\mid as\mid a\mid ons\mid ez\mid ont)$ \hfill sur $\Sigma_2=\{a,\dots,z\}$
    \item $((a\mid b\mid c)(a\mid b\mid c)(a\mid b\mid c))^\star$ \hfill sur $\Sigma_3=\{a,b,c\}$
    \item $(b\mid c)^\star a(b\mid c)^\star a(b\mid c)^\star$ \hfill sur $\Sigma_3=\{a,b,c\}$
    \end{enumerate}

  \item Donner une expression rationnelle pour chacun des langages suivants et justifier :
    \begin{enumerate}
    \item Langage $\{a^n \mid n \geq  2\}$
    \item Mots sur $\{a,b\}$ contenant exactement 2 $b$.
    \item Mots sur $\{a,b\}$ contenant un nombre pair de $a$.
    \item Mots sur $\{a,b\}$ qui commencent et finissent par le même symbole.
    \item Mots sur $\{a,b\}$ sans deux $a$ consécutifs.
    \item Écritures décimales d'entiers naturels divisibles par 5, sans $0$ inutile.
    \end{enumerate}

  \end{question}

\end{exercice}
