\begin{exercice}[Lemme de l'étoile] \label{exo:lexing/expressiveness/pumping}

  Démontrez que les langages suivants sur l'alphabet $\Sigma = \{a, b, c\}$ \emph{ne sont pas rationnels} en utilisant le lemme de l'étoile.
  \begin{question}
  \item $\left\{u \in \Sigma^\star \mid |u|_a = |u|_b + |u|_c \right\}$
  \item $\left\{a^m b^n c^p \mid m = n \lor n = p \right\}$
  \end{question}

\end{exercice}
